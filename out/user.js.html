/**
 * converts an address from an Object to a simple string drops the Geo
 * @param {Object} vector the address
 */
const addressStringArray = (vector) => {
	let tmp = {}
	Object.assign(tmp, vector)
	delete tmp.geo
	return Object.values(tmp).toString()
}
